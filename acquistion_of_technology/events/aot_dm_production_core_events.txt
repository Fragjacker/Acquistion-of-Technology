# Gamble event to determine the outcome of the experiment.
country_event = {
    id = aot_dm_production_core_events.1000
    hide_window = yes
    is_triggered_only = yes
    immediate = {
        random_situation = {
            limit = { is_situation_type = situation_aot_production_core }
            save_event_target_as = ongoing_production_core_situation
        }
        if = {
            limit = { check_variable = { which = current_dm_production_core_energy_level value = 0 } }
            event_target:ongoing_production_core_situation = {
                add_situation_progress = @dm_yield_0
            }
            country_event = {
                id = aot_dm_production_core_events.3
            }
        }
        else_if = {
            limit = { check_variable = { which = current_dm_production_core_energy_level value = 1 } }
            roll_production_core_result = { SUCCESS_CHANCE = 95 FAIL_CHANCE = 5 PROGRESS = @dm_yield_1 REGRESS = @[ - dm_yield_1 ] }
        }
        else_if = {
            limit = { check_variable = { which = current_dm_production_core_energy_level value = 2 } }
            roll_production_core_result = { SUCCESS_CHANCE = 90 FAIL_CHANCE = 10 PROGRESS = @dm_yield_2 REGRESS = @[ - dm_yield_2 ] }
        }
        else_if = {
            limit = { check_variable = { which = current_dm_production_core_energy_level value = 3 } }
            roll_production_core_result = { SUCCESS_CHANCE = 85 FAIL_CHANCE = 15 PROGRESS = @dm_yield_3 REGRESS = @[ - dm_yield_3 ] }
        }
        else_if = {
            limit = { check_variable = { which = current_dm_production_core_energy_level value = 4 } }
            roll_production_core_result = { SUCCESS_CHANCE = 80 FAIL_CHANCE = 20 PROGRESS = @dm_yield_4 REGRESS = @[ - dm_yield_4 ] }
        }
        else = {
            roll_production_core_result = { SUCCESS_CHANCE = 75 FAIL_CHANCE = 25 PROGRESS = @dm_yield_5 REGRESS = @[ - dm_yield_5 ] }
        }
    }
}
