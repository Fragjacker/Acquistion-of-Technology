namespace = acot_exterminatus_event
# Planet is turned into tomb world by Exterminatus bombing stance
planet_event = {
	id = acot_exterminatus_event.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = {
			has_global_flag = acot_exterminatus_forbidden
		}
		OR = {
			has_orbital_bombardment_stance = exterminatus
			has_orbital_bombardment_stance = punishment
		}
		OR = {
			is_memeable_planet = yes
			is_memeable_space_habitat = yes
			is_memeable_ringworld = yes
		}
	}
	immediate = {
		add_threat = {
			who = from
			amount = 10
		}
		every_owned_pop = {
			kill_pop = yes
		}
		remove_all_buildings = yes
		reroll_deposits = yes
		clear_planet_modifiers = yes
		if = {
			limit = {
				NOT = {
					has_global_flag = fractured_happened
				}
			}
			set_global_flag = fractured_happened
		}
		### FRACTURE
		planet_owner = {
			add_static_war_exhaustion = {
				attacker = from.fleet.owner
				location = root
				value_for_planet_destruction = 1
			}
			IF = {
				limit = {
					NOT = {
						has_global_flag = fractured_happened
					}
				}
				country_event = {
					id = acot_exterminatus_event.101
				}
			}
		}
		from = {
			IF = {
				limit = {
					NOT = {
						has_global_flag = fractured_happened
					}
				}
				country_event = {
					id = acot_exterminatus_event.102
				}
			}
		}
		every_playable_country = {
			limit = {
				NOR = {
					is_same_value = root.owner
					is_same_value = from
					has_global_flag = fractured_happened
				}
			}
			country_event = {
				id = acot_exterminatus_event.103
			}
		}
		if = {
			limit = {
				is_memeable_planet = yes
			}
			random_list = {
				10 = {
					modifier = {
						factor = 0
						check_variable = {
							which = acot_planet_damage
							value >= 10
						}
					}
					change_pc = pc_dark_fractured
					planet_event = {
						id = acot_exterminatus_event.1000
					}
				}
				40 = {
					modifier = {
						factor = 0
						check_variable = {
							which = acot_planet_damage
							value >= 30
						}
					}
					change_pc = pc_dark_fractured_unstable
				}
				25 = {
					modifier = {
						factor = 0
						check_variable = {
							which = acot_planet_damage
							value <= 10
						}
					}
					change_pc = pc_dark_fractured_ae
				}
				25 = {
					modifier = {
						factor = 0
						check_variable = {
							which = acot_planet_damage
							value <= 30
						}
					}
					modifier = {
						factor = 2
						check_variable = {
							which = acot_planet_damage
							value >= 60
						}
					}
					change_pc = pc_dark_fractured_ae_unstable
				}
			}
		}
		else_if = {
			limit = {
				is_memeable_ringworld = yes
			}
			change_pc = pc_ringworld_habitable_damaged
			spawn_ringworld_cracker_effect = yes
			reset_planet = yes
		}
		else_if = {
			limit = {
				is_memeable_space_habitat = yes
			}
			spawn_habitat_cracker_effect = yes
			remove_planet = yes
		}
	}
}

# Planet is turned into meme after Final Destination
planet_event = {
	id = acot_exterminatus_event.4
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = {
			has_global_flag = acot_exterminatus_forbidden
		}
		from = {
			owner = {
				OR = {
					is_country_type = primitive
					is_playable_country = yes
				}
			}
		}
		has_orbital_bombardment_stance = final_destionation
		planet_devastation >= 100
		OR = {
			is_memeable_planet = yes
			is_memeable_space_habitat = yes
			is_memeable_ringworld = yes
		}
		planet_owner = {
			or = {
				is_country_type = default
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
				is_country_type = awakened_marauders
				is_playable_country = yes
			}
		}
	}
	immediate = {
		save_event_target_as = showered_planet
		add_threat = {
			who = from
			amount = 10
		}
		every_owned_pop = {
			kill_pop = yes
		}
		planet_owner = {
			add_static_war_exhaustion = {
				attacker = from.fleet.owner
				location = root
				value_for_planet_destruction = 1
			}
			country_event = {
				id = acot_exterminatus_event.6
			}
		}
		from = {
			country_event = {
				id = acot_exterminatus_event.5
			}
		}
		remove_all_buildings = yes
		reroll_deposits = yes
		destroy_colony = yes
		reset_planet = yes
		clear_planet_modifiers = yes
		if = {
			limit = {
				is_memeable_planet = yes
			}
			change_pc = pc_shattered
		}
		else = {
			if = {
				limit = {
					is_memeable_ringworld = yes
				}
				change_pc = pc_ringworld_habitable_damaged
				spawn_ringworld_cracker_effect = yes
			}
			if = {
				limit = {
					is_memeable_space_habitat = yes
				}
				spawn_habitat_cracker_effect = yes
				remove_planet = yes
			}
		}
	}
}
