    id = acot_planets.11
    hide_window = yes

    is_triggered_only = yes

    }

    immediate = {
        if = {
            limit = {
                has_building = building_acot_ruined_absent_fortress
            }
            remove_building = building_acot_ruined_absent_fortress

                id = acot_final_sparks.106
            }

            switch = {
                trigger = has_fleet_flag

                acot_gigafortress_ruined = { add_building = building_acot_ruined_giga_fortress }

                aot_phanon_gigafortress_lab = { add_building = building_acot_ruined_giga_fortress }

            delete_fleet = this
        }
    }
}

carrier_event = {
    id = acot_planets.800
    hide_window = yes
    is_triggered_only = yes
    immediate = {
        remove_capital_building = yes
        if = {
            limit = {
                owner? = {
                    is_fallen_empire = no
                    OR = {
                        has_policy_flag = acot_capital_settlement_base_max
                        has_policy_flag = acot_capital_settlement_delta
                    }
                    has_stellarite_capital_building_tech = yes
                }
            }
            aot_add_stellarite_capital_building = yes
        }
        else_if = {
            limit = {
                owner? = {
                    is_fallen_empire = no
                    OR = {
                        has_policy_flag = acot_capital_settlement_base_max
                        has_policy_flag = acot_capital_settlement_delta
                    }
                    has_phanon_capital_building_tech = yes
                }
            }
            aot_add_phanon_capital_building = yes
        }
        else_if = {
            limit = {
                owner? = {
                    is_fallen_empire = no
                    OR = {
                        has_policy_flag = acot_capital_settlement_base_max
                        has_policy_flag = acot_capital_settlement_delta
                    }
                    has_alpha_capital_building_tech = yes
                }
            }
            aot_add_alpha_capital_building = yes
        }
        else_if = {
            limit = {
                owner? = {
                    OR = {
                        has_policy_flag = acot_capital_settlement_base_max
                        has_policy_flag = acot_capital_settlement_delta
                    }
                }
            }
            add_advanced_capital_building = yes
        }
        else = {
            add_base_capital_building = yes
        }
    }
}
