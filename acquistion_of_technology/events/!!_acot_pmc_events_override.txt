		trigger = {

	is_triggered_only = yes
	trigger = {
		fromfrom = {
			is_planet_class = pc_acot_pmc_vault
		}
	}
	immediate = {
		FROMFROM = {
			every_owned_pop = {
				limit = {
					species = {
						is_same_value = event_target:phanon_species

				count = 5

					}
				}

			}
		}
	}
	option = {
		name = acot_pmc.11.a
		trigger = {
			NOT = {
				has_country_flag = acot_first_time_vault_capture
			}
		}

				}
			}

			}
