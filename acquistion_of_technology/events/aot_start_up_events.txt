# Set global flag to detect if this mod is installed
event = {
    id = aot_startup.1
    hide_window = yes
    fire_only_once = yes
    is_triggered_only = yes
    trigger = {
        NOT = {
            has_global_flag = has_aot_mod
            has_global_flag = aot_live_version_v6
        }
    }
    immediate = {
        set_global_flag = has_aot_mod
        set_global_flag = aot_live_version_v6
    }
}
