aot_count_megaturrets_for_upgrade = {
    complex_trigger_modifier = {
        trigger = count_megastructure
        trigger_scope = solar_system
        parameters = {
            limit = {
                is_upgradeable_megaturret = yes
            }
        }
        mode = add
        [[MULT] mult = $MULT$ ]
    }
}
aot_count_megaturrets_for_retrofit = {
    complex_trigger_modifier = {
        trigger = count_ship_in_system
        trigger_scope = solar_system
        parameters = {
            limit = {
                is_ship_size = aot_megaturret
            }
        }
        mode = add
        [[MULT] mult = $MULT$ ]
    }
}
