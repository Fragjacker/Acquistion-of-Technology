inline_script = {
	script = shared/normal_buildings_base
	ALLOW = "$ALLOW$"
	CAN_BUILD = $CAN_BUILD$
	PREREQUISITE = $PREREQUISITE$
	UPGRADES_INTO = $UPGRADES_INTO$
	AI_FACTORS = "$AI_FACTORS$"
	TIER = $TIER$
}
@num_jobs = @[ 200 * $TIER$ ]
category = manufacturing
building_sets = {
	industrial
	foundry
}
potential = {
	exists = planet_owner
	planet_owner = {
		has_technology = $PREREQUISITE$
	}
}
triggered_planet_modifier = {
	potential = {
		exists = planet_owner
		planet_owner = {
			is_hive_empire = yes
			has_technology = $PREREQUISITE$
		}
	}
	modifier = {
		job_$JOB_HIVE$_add = @num_jobs
	}
}
triggered_planet_modifier = {
	potential = {
		exists = planet_owner
		planet_owner = {
			is_machine_empire = yes
		}
	}
	modifier = {
		job_$JOB_MACHINE$_add = @num_jobs
	}
}
triggered_planet_modifier = {
	potential = {
		exists = planet_owner
		planet_owner = {
			is_individualistic_empire = yes
		}
	}
	modifier = {
		job_$JOB$_add = @num_jobs
	}
}
