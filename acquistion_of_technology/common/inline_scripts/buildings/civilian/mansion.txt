	government

	entertainment
}

	NOR = {
		has_modifier = slave_colony
		has_modifier = penal_colony
	}
	planet_owner = {
		is_regular_empire = yes
	}
	NOT = { has_building = building_low_tech_capital }
	OR = {
		has_planet_flag = ignore_ai_building_limitations
		planet_owner = {
			NOT = { is_ai = yes }
		}

			NOT = { has_available_jobs = acot_entertainer }
			NOT = { has_forbidden_jobs = acot_entertainer }
		}
	}
}

	exists = planet_owner
	OR = {
		planet_owner = { is_regular_empire = no }

	}
}

	planet_housing_add = @[ auto_housing_dm * $TIER$ ]
