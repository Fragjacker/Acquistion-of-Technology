inline_script = {
	script = shared/normal_buildings_base
	ALLOW = $ALLOW$
	CAN_BUILD = $CAN_BUILD$
	PREREQUISITE = $PREREQUISITE$
	UPGRADES_INTO = $UPGRADES_INTO$
	AI_FACTORS = $AI_FACTORS$
	TIER = $TIER$
}
category = pop_assembly
building_sets = {
	government
	urban
}
destroy_trigger = {
	exists = planet_owner
	OR = {
		planet_owner = {
			OR = {
				is_regular_empire = no
				is_robot_empire = no
				has_policy_flag = robots_outlawed
			}
		}
		AND = {
			planet_owner = {
				is_ai = yes
			}
			free_district_slots = 0
			free_building_slots = 0
			free_housing <= 0
			free_jobs <= 0
		}
		planet_owner = {
			is_ai = yes
			has_ascension_perk = ap_engineered_evolution
			is_materialist = no
		}
	}
}
triggered_planet_modifier = {
	potential = {
		exists = planet_owner
		planet_owner = {
			OR = {
				is_regular_empire = yes
				is_individual_machine = yes
			}
		}
	}
	job_$JOB$_add = @[ 200 * $TIER$ * $TIER$ ]
}

	potential = {

		planet_owner = {
			is_machine_empire = yes
		}
	}
	job_$JOB_MACHINE$_add = @[ 300 * $TIER$ * $TIER$ ]
}
