	CAN_BUILD = $CAN_BUILD$

	UPGRADES_INTO = $UPGRADES_INTO$

}

	planet_owner = {

		has_technology = $PREREQUISITE$
	}

	exists = planet_owner

			is_regular_empire = no
		}
	}

	}
	upkeep = {
		$UPKEEP$
	}
}

	planet = {
		set_timed_planet_flag = {
			flag = recent_admin_building_built
			years = 2
		}
	}
