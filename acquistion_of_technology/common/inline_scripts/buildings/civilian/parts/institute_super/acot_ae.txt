triggered_planet_modifier = {
    potential = {
        owner? = {
            is_regular_empire = yes
        }
    }
    modifier = {
        pop_acot_researcher_bonus_workforce_mult = @[ $BASE_MULT$ * two_exp$EXPONENT$ ]
        job_acot_researcher_add = $BASE_AMOUNT$
    }
}
