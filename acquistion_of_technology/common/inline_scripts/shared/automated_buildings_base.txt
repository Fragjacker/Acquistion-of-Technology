base_buildtime = @[ b1_time + ( 120 * $TIER$ ) ]
can_build = $CAN_BUILD$
can_demolish = yes
allow = {
    custom_tooltip = {
        fail_text = ancrel.10004.failtext
        exists = planet_owner
        planet_owner = {
            has_technology = $PREREQUISITE$
        }
    }
    $ALLOW$
}
upgrades = {
    $UPGRADES_INTO$
}
prerequisites = {
    $PREREQUISITE$
}
show_in_tech = $PREREQUISITE$
resources = {
    category = planet_buildings
    cost = {
        $COST$
    }
    produces = {
        $PRODUCES$
    }
    upkeep = {
        $UPKEEP$
    }
    cost = {
        trigger = {
            owner = { is_wilderness_empire = yes }
        }
        biomass = @[ b3_biomass * $TIER$ ]
    }
}
triggered_planet_modifier = {
    potential = {
        exists = owner
        owner = {
            is_wilderness_empire = yes
            has_tradition = tr_purity_finish
        }
    }
    planet_building_build_speed_mult = @[ 0.03 * $TIER$ ]
    planet_decision_enact_speed_mult = @[ 0.03 * $TIER$ ]
}
on_queued = {
    if = {
        # kill on the same planet
        limit = {
            is_variable_set = local_spent_biomass
            any_owned_pop_group = {
                has_trait = trait_wilderness
                pop_group_size > local_spent_biomass
            }
        }
        random_owned_pop_group = {
            limit = {
                has_trait = trait_wilderness
                pop_group_size > local_spent_biomass
            }
            kill_pop_group = {
                pop_group = this
                amount = local_spent_biomass
            }
        }
    }
    else_if = {
        limit = { is_variable_set = local_spent_biomass }
        owner = {
            global_biomass_spender = yes
        }
    }
}
on_unqueued = {
    unqueue_biomass_effect = yes
}
ai_weight = {
    weight = 1000
    $AI_FACTORS$
}
