#------------------<  Base inline script for jobs that produce just one resource.  >----------------
inline_script = {
	script = pop_jobs/part/$CATEGORY$
	CATEGORY = $CATEGORY$
	JOB = $JOB$
	RESOURCE = $RESOURCE$
}
resources = {
	category = planet_jobs_productive
	produces = {
		$RESOURCE$ = $PRODUCTION_AMOUNT$
	}
	produces = {
		trigger = {
			planet = {
				is_planet_class = pc_acot_enigmopolis
			}
		}
		$RESOURCE$ = @[ $PRODUCTION_AMOUNT$ * 4 ]
	}
	produces = {
		trigger = {
			planet = {
				OR = {
					is_planet_class = pc_dark_fractured
					is_planet_class = pc_acot_enigmatic
				}
			}
		}
		$RESOURCE$ = $PRODUCTION_AMOUNT$
	}
	produces = {
		trigger = {
			planet = {
				has_modifier = $TRIGGER_MODIFIER$
			}
		}
		$RESOURCE$ = $PRODUCTION_AMOUNT$
	}
	produces = {
		trigger = {
			planet = {
				OR = {
					is_planet_class = pc_dark_fractured
					is_planet_class = pc_acot_enigmatic
				}
				has_modifier = $TRIGGER_MODIFIER$
			}
		}
		$RESOURCE$ = $PRODUCTION_AMOUNT$
	}
	upkeep = {
		trigger = {
			planet = {
				has_modifier = $TRIGGER_MODIFIER$
			}
		}
		$UPKEEP_RESOURCE$ = $UPKEEP_AMOUNT$
	}
}
inline_script = "jobs/automodding_priority_$RESOURCE$"
promotion = {
	time = @standard_promotion_time
	effect = {
		pop_amount_change_category = {
			CATEGORY = $CATEGORY$
			AMOUNT = local_pop_amount
		}
	}
}
