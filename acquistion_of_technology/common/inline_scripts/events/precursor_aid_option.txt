### Precursor Databank option to provide techs to allies.
option = {
    name = $NAME$
    allow = {
        has_technology = $PREREQUISITE$
    }
    trigger = {
        any_country = {
            OR = {
                has_defensive_pact = root
                is_in_federation_with = root
                root = {
                    is_overlord_to = prev
                }
            }
            NOR = {
                has_technology = $PREREQUISITE$
                has_tech_option = $PREREQUISITE$
            }
        }
    }
    hidden_effect = {
        remove_country_flag = acot_databank_diplomacy_engaged
        country_event = { id = $AID_EVENT$ }
    }
}
