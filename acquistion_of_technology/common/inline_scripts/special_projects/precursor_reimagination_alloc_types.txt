cost = @[ 500000 * two_exp$EXPONENT$ ]
tech_department = physics_technology
picture = GFX_evt_physics_research
event_chain = acot_precursor_design_chain
event_scope = country_event
location = no
abort_trigger = {
}
on_success = {
    owner = {
        add_event_chain_counter = {
            event_chain = acot_precursor_design_chain
            counter = attempts_remain
            amount = -1
        }
        change_variable = {
            which = acot_quota_attempts
            value = -1
        }
        random_list = {
            60 = {
                add_event_chain_counter = {
                    event_chain = acot_precursor_design_chain
                    counter = progress_numbers
                    amount = @[ 100 * two_exp$EXPONENT$ ]
                }
                change_variable = {
                    which = acot_quota_required
                    value = @[ - 100 * two_exp$EXPONENT$ ]
                }
            }
            20 = {
                add_event_chain_counter = {
                    event_chain = acot_precursor_design_chain
                    counter = progress_numbers
                    amount = @[ 200 * two_exp$EXPONENT$ ]
                }
                change_variable = {
                    which = acot_quota_required
                    value = @[ - 200 * two_exp$EXPONENT$ ]
                }
            }
            20 = {
                add_event_chain_counter = {
                    event_chain = acot_precursor_design_chain
                    counter = progress_numbers
                    amount = @[ 50 * two_exp$EXPONENT$ ]
                }
                change_variable = {
                    which = acot_quota_required
                    value = @[ - 50 * two_exp$EXPONENT$ ]
                }
            }
        }
        ### SUCCESS !!?!?!?!
        if = {
            limit = {
                check_variable = {
                    which = acot_quota_required
                    value <= 0
                }
            }
            country_event = {
                id = acot_precursor_databank.904
            }
        }
        ### INCOMPLETED, BUT STILL HAS ATTEMPTS
        else_if = {
            limit = {
                check_variable = {
                    which = acot_quota_attempts
                    value > 0
                }
                check_variable = {
                    which = acot_quota_required
                    value > 0
                }
            }
            country_event = {
                id = acot_precursor_databank.902
            }
        }
        ### INCOMPLETED, NO ATTEMPTS LEFT
        else_if = {
            limit = {
                check_variable = {
                    which = acot_quota_attempts
                    value <= 0
                }
                check_variable = {
                    which = acot_quota_required
                    value > 0
                }
            }
            country_event = {
                id = acot_precursor_databank.903
            }
        }
    }
}
on_fail = {
}
