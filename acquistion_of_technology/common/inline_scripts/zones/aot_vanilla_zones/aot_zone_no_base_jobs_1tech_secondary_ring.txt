###########################
# This inline script makes all 4 versions of the districts with a consistent naming scheme
###########################

zone_aot_DM_$TYPE$_ring_world = {
    icon = GFX_xhk_aot_zone_DM_$GFX_SUFFIX$
    inline_script = {
        script = zones/aot_vanilla_zones/base/aot_zone_DM_base
        OWNER_REQ = "
            has_technology = tech_acot_dm_$TECH_SUFFIX$
            $OTHER_OWNER_REQ$
        "
        PLANET_REQ = "$PLANET_REQ$"
    }

    inline_script = {
        script = zones/aot_general_zone_job_adder
        NORMAL_JOB = $DM_NORM_JOB$
        GESTALT_JOB = $DM_GEST_JOB$
        AMOUNT = $AMOUNT$
    }

    swap_type = district_aot_DM_$TYPE$_ring_world
    swap_type_weight = 10

    included_building_sets = {
        $BUILDING_SETS$
    }

    zone_sets = {
        ring_world
    }

    inline_script = {
        script = zones/shared_ring_world_zone_modifiers
    }
}
