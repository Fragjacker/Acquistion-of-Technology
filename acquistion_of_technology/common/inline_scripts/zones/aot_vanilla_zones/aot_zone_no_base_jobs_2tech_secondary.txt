###########################
# This inline script makes all 4 versions of the districts with a consistent naming scheme
###########################

zone_aot_PE_$TYPE$ $TRAILER_TYPE_LEADING_UNDERSCORE$ = {
    icon = GFX_xhk_aot_zone_PE_$GFX_SUFFIX$
    inline_script = {
        script = zones/aot_vanilla_zones/base/aot_zone_PE_base
        OWNER_REQ = "
            has_technology = tech_phanon_$TECH_SUFFIX$
            has_technology = tech_phanon_$TECH_SUFFIX2$
            $OTHER_OWNER_REQ$
        "
        PLANET_REQ = "$PLANET_REQ$"
    }

    inline_script = {
        script = zones/aot_general_zone_job_adder
        NORMAL_JOB = $PE_NORM_JOB$
        GESTALT_JOB = $PE_GEST_JOB$
        AMOUNT = $AMOUNT$
    }
    inline_script = {
        script = zones/aot_general_zone_job_adder
        NORMAL_JOB = $PE_NORM_JOB2$
        GESTALT_JOB = $PE_GEST_JOB2$
        AMOUNT = $AMOUNT$
    }

    swap_type = district_aot_PE_$TYPE$ $TRAILER_TYPE_LEADING_UNDERSCORE$
    swap_type_weight = 10

    included_building_sets = {
        $BUILDING_SETS$
    }

    zone_sets = {
        $ZONE_SETS$
    }

    inline_script = {
        script = zones/shared_arcology_zone_modifiers
    }
}
