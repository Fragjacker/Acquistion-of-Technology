is_crackable_celestial_object = {
    optimize_memory
    OR = {
        uninhabitable_regular_planet = yes
        is_memeable_planet = yes
        is_memeable_ringworld = yes
        is_memeable_space_habitat = yes
    }
}

is_memeable_planet = {
    optimize_memory
    OR = {
        is_giga_planet = yes
        pd_is_planet_class_all = yes
        AND = {
            colonizable_planet = yes
            is_artificial = no
        }
    }
}

is_shielded_planet = {
    optimize_memory
    OR = {
        has_active_building = building_phanon_psg
        has_active_building = building_stellarite_psg
        is_planet_class = pc_shielded
        is_planet_class = pc_ringworld_shielded
        is_planet_class = pc_habitat_shielded
        giga_is_birch_world = yes
    }
}

can_destroy_planet_with_AoT_PLANET_KILLER_CRACKER = {
    custom_tooltip = {
        fail_text = is_not_a_standard_planet_or_megastructure
        is_a_planet_megastructure = yes
    }
    custom_tooltip = {
        fail_text = is_shielded
        is_shielded_planet = no
    }
    custom_tooltip = {
        fail_text = fotd_cant_collossus
        NOT = { has_carrier_flag = fotd_seperatist_planet@from.owner }
        from.owner? = {
            NOT = { has_country_flag = fotd_seperatist_country@from.owner }
        }
    }
    OR = {
        custom_tooltip = {
            text = is_not_hostile_cracker
            OR = {
                AND = {
                    space_owner? = {
                        OR = {
                            is_at_war_with = from.owner
                            is_country_type = swarm
                            is_country_type = extradimensional
                            is_country_type = extradimensional_2
                            is_country_type = extradimensional_3
                            is_country_type = ai_empire
                        }
                    }
                }
                AND = {
                    # colony
                    owner? = {
                        OR = {
                            is_at_war_with = from.owner
                            is_country_type = swarm
                            is_country_type = extradimensional
                            is_country_type = extradimensional_2
                            is_country_type = extradimensional_3
                            is_country_type = ai_empire
                        }
                    }
                }
            }
        }
        AND = {
            # primitive target
            custom_tooltip = {
                text = cannot_crack_pre_ftl_planet_ownership
                owner? = {
                    is_primitive = yes
                }
                space_owner? = {
                    OR = {
                        is_same_value = from.owner
                        is_at_war_with = from.owner
                        is_country_type = swarm
                        is_country_type = extradimensional
                        is_country_type = extradimensional_2
                        is_country_type = extradimensional_3
                        is_country_type = ai_empire
                    }
                }
            }
            custom_tooltip = {
                text = cannot_crack_pre_ftl_planet_interference
                # primitive
                owner? = {
                    is_primitive = yes
                }
                from.owner? = {
                    has_policy_flag = interference_aggressive
                }
            }
        }
        custom_tooltip = {
            text = not_barren_molten_frozen_toxic_test_fire
            NOT = {
                exists = owner
            }
            OR = {
                NOT = {
                    exists = space_owner
                }
                AND = {
                    space_owner? = { is_same_value = from.owner }
                }
            }
        }
        is_crackable_celestial_object = yes
    }
}
