aot_add_advanced_capital_building = {
    IF = {
        limit = {
            owner = {
                is_regular_empire = yes
            }
        }
        add_building = $CAPITAL_REGULAR$
    }
    else_if = {
        limit = {
            owner = {
                is_hive_empire = yes
            }
        }
        add_building = $CAPITAL_HIVE$
    }
    else_if = {
        limit = {
            owner = {
                is_machine_empire = yes
            }
        }
        add_building = $CAPITAL_MACHINE$
    }
}
aot_add_stellarite_capital_building = {
    aot_add_advanced_capital_building = {
        CAPITAL_REGULAR = building_stellarite_palace
        CAPITAL_HIVE = building_stellarite_hive_cluster
        CAPITAL_MACHINE = building_stellarite_control_center
    }
}

aot_spawn_giga_fortress_in_space = {
    create_fleet = {
        effect = {
            set_owner = from.owner
            create_ship = {
                name = random
                design = "$DESIGN$"
                graphical_culture = from.owner
            }
            set_location = {
                target = prev
                distance = 30
                angle = random
            }
        }
        settings = {
            can_upgrade = no
        }
    }
    [[FLEET_FLAG]
        last_created_fleet = {
            set_fleet_flag = $FLEET_FLAG$
        }
    ]
}
