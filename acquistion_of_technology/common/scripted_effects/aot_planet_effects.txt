aot_add_advanced_capital_building = {
    IF = {
        limit = {
            owner = {
                is_regular_empire = yes
            }
        }
        add_building = $CAPITAL_REGULAR$
    }
    else_if = {
        limit = {
            owner = {
                is_hive_empire = yes
            }
        }
        add_building = $CAPITAL_HIVE$
    }
    else_if = {
        limit = {
            owner = {
                is_machine_empire = yes
            }
        }
        add_building = $CAPITAL_MACHINE$
    }
}
aot_add_stellarite_capital_building = {
    aot_add_advanced_capital_building = {
        CAPITAL_REGULAR = building_stellarite_palace
        CAPITAL_HIVE = building_stellarite_hive_cluster
        CAPITAL_MACHINE = building_stellarite_control_center
    }
}
